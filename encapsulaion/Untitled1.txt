class MyArrayList {
    private int[] arr;
    private int size;

    // Constructor
    public MyArrayList() {
        arr = new int[10]; // default capacity
        size = 0;
    }

    // Add element
    public void add(int data) {
        if (size == arr.length) {
            grow();
        }
        arr[size] = data;
        size++;
    }

    // Grow method (resize array)
    private void grow() {
        int newSize = arr.length + (arr.length / 2); // 1.5x growth
        int[] newArr = new int[newSize];

        for (int i = 0; i < arr.length; i++) {
            newArr[i] = arr[i];
        }

        arr = newArr;
    }

    // Get element
    public int get(int index) {
        if (index >= size) {
            throw new IndexOutOfBoundsException();
        }
        return arr[index];
    }

    // Remove element
    public void remove(int index) {
        if (index >= size) {
            throw new IndexOutOfBoundsException();
        }

        for (int i = index; i < size - 1; i++) {
            arr[i] = arr[i + 1];
        }

        size--;
    }

    // Display
    public void display() {
        for (int i = 0; i < size; i++) {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
    }
}
